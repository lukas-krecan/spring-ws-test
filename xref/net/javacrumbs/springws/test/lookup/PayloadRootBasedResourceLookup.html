66  	public Resource lookupResource(URI uri, WebServiceMessage message) throws IOException {
67  		QName payloadQName;
68  		try {
69  			payloadQName = PayloadRootUtils.getPayloadRootQName(message.getPayloadSource(), TRANSFORMER_FACTORY);
70  		} catch (TransformerException e) {
71  			logger.warn("Can not resolve payload.",e);
72  			return null;
73  		} catch (XMLStreamException e) {
74  			logger.warn("Can not resolve payload.",e);
75  			return null;
76  		}
77  		String payloadName = payloadQName.getLocalPart();
78  		String[] expressions = getDiscriminators(payloadName);
79  		Document document = getXmlUtil().loadDocument(message);
80  		Resource resource; 
81  		int discriminatorsCount = expressions.length;
82  		do
83  		{
84  			String resourceName = getResourceName(uri, payloadName, expressions, discriminatorsCount, document);
85  			logger.debug("Looking for resource "+resourceName);
86  			resource = getResourceLoader().getResource(resourceName);
87  			discriminatorsCount--;
88  		}
89  		while((resource == null || !resource.exists()) && discriminatorsCount>=0);
90  		if (resource!=null && resource.exists())
91  		{
92  			logger.debug("Found resource "+resource);
93  			return processResource(uri, message, resource);
94  		}
95  		else
96  		{
97  			return null;
98  		}
99  	}

102 	private String[] getDiscriminators(String payloadName) {
103 		String[] result = discriminators.get(payloadName);
104 		return result!=null?result:new String[0];
105 	}
106 	
107 	protected String getResourceName(URI uri, String payloadName, String[] expressions, int discriminatorsCount, Document document) {
108 		String uriHost = prependUri?(uri.getHost()+payloadDelimiter):"";
109 		return pathPrefix+uriHost+payloadName+payloadDelimiter+getDiscriminatorExpression(uri, expressions, discriminatorsCount, document)+pathSuffix;
110 	}
111 	/**
112 	 * Returns expression generated from discriminators
113 	 * @param uri
114 	 * @param expressions
115 	 * @param discriminatorsCount 
116 	 * @param document
117 	 * @return
118 	 */
119 	protected String getDiscriminatorExpression(URI uri, String[] expressions, int discriminatorsCount, Document document) {
120 		StringBuilder result = new StringBuilder();
121 		for (int i=0; i<discriminatorsCount;i++) {
122 			String expression = expressions[i];
123 			String evaluated = evaluateExpression(expression, uri, document);
124 			if (StringUtils.hasText(evaluated))
125 			{
126 				result.append(evaluated).append(discriminatorDelimiter);
127 			}
128 		}	 
129 		return result.toString(); 
130 	}
