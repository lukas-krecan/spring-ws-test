37  /**
38   * Validates message using XML schema.
39   * @author Lukas Krecan
40   *
41   */
42  public class SchemaValidator {
43      protected final Log logger = LogFactory.getLog(getClass());
44  	
45  	public void validate(WebServiceMessage message, XmlValidator validator) throws IOException{
46  		Source requestSource = message.getPayloadSource();
47          if (requestSource != null) {
48              SAXParseException[] errors = validator.validate(requestSource);
49              if (!ObjectUtils.isEmpty(errors)) {
50                  handleRequestValidationErrors(message, errors);
51              }
52              logger.debug("Request message validated");
53          }
54          else
55          {
56         	 logger.warn("Request source is null");
57          }
58  	}
59  	
60  	/**
61  	 * Creates {@link XmlValidator} from schemas.
62  	 * @param schemas
63  	 * @param schemaLanguage
64  	 * @return
65  	 * @throws IOException
66  	 */
67  	public XmlValidator createValidatorFromSchemas(Resource[] schemas, String schemaLanguage) throws IOException
68  	{
69  		 Assert.hasLength(schemaLanguage, "schemaLanguage is required");
70           for (int i = 0; i < schemas.length; i++) {
71               Assert.isTrue(schemas[i].exists(), "schema [" + schemas[i] + "] does not exist");
72           }
73           if (logger.isInfoEnabled()) {
74               logger.info("Validating using \"" + StringUtils.arrayToCommaDelimitedString(schemas)+"\"");
75           }
76           return XmlValidatorFactory.createValidator(schemas, schemaLanguage);
77  	}
78  	
79  	
80      protected void handleRequestValidationErrors(WebServiceMessage message, SAXParseException[] errors) {
81  		throw new WsTestException("Request not valid. "+Arrays.toString(errors),message);
82  	}
83  }
