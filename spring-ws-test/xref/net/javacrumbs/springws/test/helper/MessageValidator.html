95  	/**
96  	 * Validates if the message corresponds to given XSDs.
97  	 * @param message
98  	 */
99  	public MessageValidator validate(String schemaPath, String... schemaPaths) throws IOException {
100 		Resource[] schemas = new Resource[schemaPaths.length];
101 		for (int i = 0; i < schemas.length; i++) {
102 			schemas[i] = resourceLoader.getResource(schemaPaths[i]);
103 		}
104 		return validate(resourceLoader.getResource(schemaPath), schemas);
105 	}

107 	/**
108 	 * Validates if the message corresponds to given XSD.
109 	 * @param message
110 	 */
111 	public MessageValidator validate(Resource schema, Resource... schemas) throws IOException {
112 		
113 		Resource[] joinedSchemas = new Resource[schemas.length+1];
114 		joinedSchemas[0] = schema;
115 		System.arraycopy(schemas, 0, joinedSchemas, 1, schemas.length);
116 		
117 		validate(schemaValidator.createValidatorFromSchemas(joinedSchemas, schemaLanguage ));
118 		return this;
119 	}

358 	/**
359 	 * Returns true if message contains regexp.
360 	 * @param regexp
361 	 * @return 
362 	 */
363 	private boolean contains(String regexp) {
364 		String messageAsText = xmlUtil.serializeDocument(message);
365 		Pattern pattern = Pattern.compile(regexp);
366 		return pattern.matcher(messageAsText).find();
367 	}
