52  		String result = resolveXPath(xpath);

66  		String xpath = "concat($context.testName,'-response.xml')";
67  		String result = resolveXPath(xpath);
68  		assertEquals("test136-response.xml",result);
69  	}

71  
72  	private String resolveXPath(String xpath) throws URISyntaxException, IOException {
73  		XPathExpressionResolver evaluator = new XPathExpressionResolver();
74  		Map<String, String> namespaceMap = new HashMap<String, String>();
75  		namespaceMap.put("ns", "http://www.springframework.org/spring-ws/samples/airline/schemas/messages");
76  		namespaceMap.put("soapenv", "http://schemas.xmlsoap.org/soap/envelope/");
77  		evaluator.setNamespaceMap(namespaceMap);
78  		
79  		URI uri = new URI("http://www.csa.cz");
80  		
81  		String expression = xpath;
82  		Document document = getXmlUtil().loadDocument(createMessage("xml/PRG-DUB-request.xml"));
83  		String result = evaluator.resolveExpression(expression, uri, document);
84  		return result;
85  	}
